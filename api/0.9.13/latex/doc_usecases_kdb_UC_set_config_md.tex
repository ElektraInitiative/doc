\begin{DoxyItemize}
\item {\bfseries{Precondition\+:}} -\/
\item {\bfseries{Main success scenario\+:}}
\begin{DoxyItemize}
\item Application asks to store new configuration into KDB
\item Elektra stores configuration in backends
\end{DoxyItemize}
\item {\bfseries{Alternative scenario\+:}} -\/
\item {\bfseries{Error scenario\+:}}
\begin{DoxyItemize}
\item Application asks to store new configuration into KDB
\item Elektra fails to store configuration
\item Elektra returns error with problem description
\end{DoxyItemize}
\item {\bfseries{Postcondition\+:}}
\begin{DoxyItemize}
\item The provided configuration MUST be stored into the KDB exactly. Loading the configuration later (assuming no outside modifications) MUST return the provided configuration.
\item In case of error, the KDB MUST NOT be modified. Loading the configuration later (assuming no outside modifications) MUST return the same configuration as before the attempt to store new configuration.
\end{DoxyItemize}
\item {\bfseries{Non-\/functional Constraints\+:}}
\begin{DoxyItemize}
\item Storing configuration MUST only touch the backends that have new data. 
\end{DoxyItemize}
\end{DoxyItemize}
