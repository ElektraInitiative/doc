 \hypertarget{doc_help_kdb-find_md_autotoc_md2368}{}\doxysection{SYNOPSIS}\label{doc_help_kdb-find_md_autotoc_md2368}
{\ttfamily kdb find $<$regex$>$}

Where {\ttfamily regex} is a regular expression which contains the key to find.\hypertarget{doc_help_kdb-find_md_autotoc_md2369}{}\doxysection{DESCRIPTION}\label{doc_help_kdb-find_md_autotoc_md2369}
This command will list the name of all keys that contain {\ttfamily regex}.\hypertarget{doc_help_kdb-find_md_autotoc_md2370}{}\doxysection{OPTIONS}\label{doc_help_kdb-find_md_autotoc_md2370}

\begin{DoxyItemize}
\item {\ttfamily -\/H}, {\ttfamily -\/-\/help}\+: Show the man page.
\item {\ttfamily -\/V}, {\ttfamily -\/-\/version}\+: Print version info.
\item {\ttfamily -\/p}, {\ttfamily -\/-\/profile $<$profile$>$}\+: Use a different kdb profile.
\item {\ttfamily -\/C}, {\ttfamily -\/-\/color $<$when$>$}\+: Print never/auto(default)/always colored output.
\item {\ttfamily -\/v}, {\ttfamily -\/-\/verbose}\+: Explain what is happening. Prints additional information in case of errors/warnings.
\item {\ttfamily -\/d}, {\ttfamily -\/-\/debug}\+: Give debug information. Prints additional debug information in case of errors/warnings.
\item {\ttfamily -\/0}, {\ttfamily -\/-\/null}\+: Use binary 0 termination.
\end{DoxyItemize}\hypertarget{doc_help_kdb-find_md_autotoc_md2371}{}\doxysection{EXAMPLES}\label{doc_help_kdb-find_md_autotoc_md2371}

\begin{DoxyCode}{0}
\DoxyCodeLine{\# Backup-\/and-\/Restore: user:/tests/find}
\DoxyCodeLine{}
\DoxyCodeLine{\# We use the `dump` plugin, since some storage plugins, e.g. INI,}
\DoxyCodeLine{\# create intermediate keys, such as `user:/tests/find/tests/foo`}
\DoxyCodeLine{\# for the following test.}
\DoxyCodeLine{sudo kdb mount find.ecf user:/tests/find dump}
\DoxyCodeLine{}
\DoxyCodeLine{\# Create the keys we use for the examples}
\DoxyCodeLine{kdb set user:/tests/find/tests val1}
\DoxyCodeLine{kdb set user:/tests/find/tests/foo/bar val2}
\DoxyCodeLine{kdb set user:/tests/find/tests/fizz/buzz fizzbuzz}
\DoxyCodeLine{kdb set user:/tests/find/tostfizz val3}
\DoxyCodeLine{kdb set user:/tests/find/tust/level lvl}
\DoxyCodeLine{}
\DoxyCodeLine{\# list all keys containing /tests/find/t[eo]}
\DoxyCodeLine{kdb find '/tests/find/t[eo]'}
\DoxyCodeLine{\#> user:/tests/find/tests}
\DoxyCodeLine{\#> user:/tests/find/tests/fizz/buzz}
\DoxyCodeLine{\#> user:/tests/find/tests/foo/bar}
\DoxyCodeLine{\#> user:/tests/find/tostfizz}
\DoxyCodeLine{}
\DoxyCodeLine{\# list all keys containing fizz}
\DoxyCodeLine{kdb find 'fizz'}
\DoxyCodeLine{\#> user:/tests/find/tests/fizz/buzz}
\DoxyCodeLine{\#> user:/tests/find/tostfizz}
\DoxyCodeLine{}
\DoxyCodeLine{kdb rm -\/r /tests/find}
\DoxyCodeLine{sudo kdb umount user:/tests/find}
