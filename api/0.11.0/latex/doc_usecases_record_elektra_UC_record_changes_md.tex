\begin{DoxyItemize}
\item {\bfseries{Title\+:}} Recording changes to the key database
\item {\bfseries{Scope\+:}} Configuration/\+Logging
\item {\bfseries{Level\+:}} User goal
\item {\bfseries{Actors\+:}} User (usually a sysadmin)
\item {\bfseries{Brief\+:}} Any changes that are made in the key database are recorded.
\end{DoxyItemize}\hypertarget{doc_usecases_record_elektra_UC_record_changes_md_autotoc_md4833}{}\doxysection{Scenarios}\label{doc_usecases_record_elektra_UC_record_changes_md_autotoc_md4833}

\begin{DoxyItemize}
\item {\bfseries{Precondition\+:}}
\begin{DoxyItemize}
\item Recording is enabled
\item An active recording session exists
\end{DoxyItemize}
\item {\bfseries{Main success scenario\+:}}
\begin{DoxyItemize}
\item User makes a change to the key database, e.\+g. add key, delete key, modify key, modify meta
\item The old \& new values for every changed key and metakey are recorded.
\end{DoxyItemize}
\item {\bfseries{Alternative scenario\+:}} -\/
\item {\bfseries{Error scenario\+:}} The {\ttfamily kdb} command will print an error message that the changes could not be recorded.
\item {\bfseries{Postcondition\+:}} All changed keys and their original values are recorded.
\item {\bfseries{Non-\/functional Constraints\+:}}
\begin{DoxyItemize}
\item The recording mechanismn should not noticably slow down Elektra. 
\end{DoxyItemize}
\end{DoxyItemize}
