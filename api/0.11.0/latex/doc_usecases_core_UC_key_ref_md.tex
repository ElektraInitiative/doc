\begin{DoxyItemize}
\item {\bfseries{Precondition\+:}}
\begin{DoxyItemize}
\item \mbox{\hyperlink{doc_usecases_core_UC_key_create_md}{\`{}\+Key\`{} has been created}}
\end{DoxyItemize}
\item {\bfseries{Main success scenario\+:}}
\begin{DoxyItemize}
\item Caller requests value of reference counter of a {\ttfamily Key}
\item Core returns current value of reference counter of a {\ttfamily Key}
\end{DoxyItemize}
\item {\bfseries{Alternative scenario\+:}}
\begin{DoxyItemize}
\item Caller requests to increment value of reference counter of a {\ttfamily Key}
\item Core returns current value of reference counter of a {\ttfamily Key} after incrementing
\end{DoxyItemize}
\item {\bfseries{Alternative scenario\+:}}
\begin{DoxyItemize}
\item Caller requests to decrement value of reference counter of a {\ttfamily Key}
\item Core returns current value of reference counter of a {\ttfamily Key} after decrementing
\end{DoxyItemize}
\item {\bfseries{Error scenario\+:}}
\begin{DoxyItemize}
\item Caller requests to increment value of reference counter of a {\ttfamily Key} whose reference counter is INT\+\_\+\+MAX
\item Reference counter will not get incremented and stays at INT\+\_\+\+MAX
\end{DoxyItemize}
\item {\bfseries{Error scenario\+:}}
\begin{DoxyItemize}
\item Caller requests to decrement value of reference counter of a {\ttfamily Key} whose reference counter is 0
\item Reference counter will not get decremented and stays at 0
\end{DoxyItemize}
\item {\bfseries{Postcondition\+:}}
\begin{DoxyItemize}
\item Reference Counter has a value from 0 to INT\+\_\+\+MAX
\end{DoxyItemize}

\end{DoxyItemize}
