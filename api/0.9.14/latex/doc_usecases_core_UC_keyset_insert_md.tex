\begin{DoxyItemize}

\item If necessary Core resizes {\ttfamily Key\+Set}
\item If necessary Core removes existing {\ttfamily Key} with same name
\item Core adds {\ttfamily Key} to {\ttfamily Key\+Set}
\end{DoxyItemize}
\item {\bfseries{Alternative scenario\+:}}
\begin{DoxyItemize}

\item If necessary Core resizes B
\item If necessary Core removes all existing {\ttfamily Key}s contained in A and B from B
\item Core copies all {\ttfamily Key}s from A to B
\end{DoxyItemize}
