\begin{DoxyItemize}
\item {\bfseries{Precondition\+:}}
\begin{DoxyItemize}
\item \mbox{\hyperlink{doc_usecases_core_UC_key_create_md}{\`{}\+Key\`{} K1 has been created}}
\item \mbox{\hyperlink{doc_usecases_core_UC_key_create_md}{\`{}\+Key\`{} K2 has been created}}
\end{DoxyItemize}
\item {\bfseries{Main success scenario\+:}}
\begin{DoxyItemize}
\item Caller requests to copy one or multiple fields from {\ttfamily Key} k1 to {\ttfamily Key} k2
\item Fields that can be copied are name, value and metadata
\item Fields, that caller wanted to copy, of {\ttfamily Key} k2 now have the value of the respective fields of {\ttfamily Key} k1
\end{DoxyItemize}
\item {\bfseries{Alternative scenario\+:}}
\item {\bfseries{Error scenario\+:}}
\begin{DoxyItemize}
\item {\ttfamily Key} k2 has field that should be copied marked as read-\/only (e.\+g., because {\ttfamily Key} is part of ordered collection like {\ttfamily Key\+Set})
\item Caller requests to copy read-\/only field in {\ttfamily Key} k2 from {\ttfamily Key} k1
\item Copying to k2 fails and k2 retains the old field values
\end{DoxyItemize}
\item {\bfseries{Postcondition\+:}}
\begin{DoxyItemize}
\item \mbox{\hyperlink{doc_usecases_core_UC_keyname_hierarchy_md}{Comparing}} the respective fields from k1 and k2 returns that those fields are equal
\end{DoxyItemize}
