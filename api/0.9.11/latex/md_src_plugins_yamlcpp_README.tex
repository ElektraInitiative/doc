\begin{DoxyCode}{0}
\DoxyCodeLine{directory      = Directory Data}
\DoxyCodeLine{directory/file = Leaf Data}

\end{DoxyCode}

\DoxyCodeLine{directory            =}

\DoxyCodeLine{directory/file       = Leaf Data}

\end{DoxyCode}

\begin{DoxyCode}{0}
\DoxyCodeLine{directory:}
\DoxyCodeLine{  \_\_\_dirdata: Directory Data}
\DoxyCodeLine{  file: Leaf Data}
